// Button.stories.ts

import type { Meta, StoryObj } from '@storybook/marko';

import Button from './Button';
import MarginDecorator from './margin-decorator.marko';

const meta: Meta<typeof Button> = {
  component: Button,
};

export default meta;
type Story = StoryObj<typeof Button>;

export const Primary: Story = {
  decorators: [
    (Story) => ({
      component: MarginDecorator,
      input: {
        renderBody: Story,
      },
    }),
  ],
};
