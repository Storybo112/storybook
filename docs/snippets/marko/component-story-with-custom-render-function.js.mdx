```js
// MyComponent.stories.js

import MyComponent from './MyComponent.marko';
import wrapper from './wrapper.marko';

export default {
  title: 'MyComponent',
  component: MyComponent,
};

// This story uses a render function to fully control how the component renders.
export const Example = {
  render: () => ({
    component: wrapper,
    input: {
      renderBody: MyComponent,
    },
  }),
};
```

```marko
// wrapper.marko

<Layout>
  <header>
    <h1>Example</h1>
  </header>
  <article>
    <${input.renderBody}>
  </article>
</Layout>
```
