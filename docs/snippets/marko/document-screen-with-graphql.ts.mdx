```marko
// YourPage.marko

import { gql } from "@marko/urql";

static const AllInfoQuery = gql`
  query AllInfo {
    user {
      userID
      name
    }
    document {
      id
      userID
      title
      brief
      status
    }
    subdocuments {
      id
      userID
      title
      content
      status
    }
  }
`

<gql-client url='https://foobar.com/v1/api' />
<gql-query query=AllInfoQuery>
  <@then|{ data, error }|>
    <if(error)>
      There was an error fetching the data!
    </if>
    <else>
      <PageLayout user={data.user}>
        <DocumentHeader document={data.document} />
        <DocumentList documents={data.subdocuments} />
      </PageLayout>
    </else>
  </@then>
  <@placeholder>
    <p>Loading...</p>
  </@placeholder>
</gql-query>
```
