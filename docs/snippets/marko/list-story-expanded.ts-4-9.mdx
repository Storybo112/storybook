// List.stories.ts

import type { Meta, StoryObj } from '@storybook/marko';

import List from './List.marko';
import ListItem from './ListItem.marko';

const meta = {
  component: List,
} satisfies Meta<typeof List>;

export default meta;
type Story = StoryObj<typeof meta>;

export const Empty: Story = {};

/*
 *ðŸ‘‡ Render functions are a framework specific feature to allow you control on how the component renders.
 * See https://storybook.js.org/docs/marko/api/csf
 * to learn how to use render functions.
 */
export const OneItem: Story = {
  render: (input) => ({
    component: List,
    input: {
      ...input,
      renderBody(out) {
        ListItem.render({}, out);
      },
    },
  }),
};

export const ManyItems: Story = {
  render: (input) => ({
    component: List,
    input: {
      ...input,
      renderBody(out) {
        ListItem.render({}, out);
        ListItem.render({}, out);
        ListItem.render({}, out);
      },
    },
  }),
};
