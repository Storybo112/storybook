import type { Meta, StoryObj } from '@storybook/marko';

import List from './List.marko';

          ListItem.render({ ...item }, out);
        }
      },
    },
  }),
};

  ...ListTemplate,
  args: {
    items: [],
  },
};

export const OneItem = {
  ...ListTemplate,
  args: {
    items: [{ ...Unchecked.args }],
  },
};
